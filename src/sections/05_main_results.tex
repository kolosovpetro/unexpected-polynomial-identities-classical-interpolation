\begin{theorem}
    \label{theorem:odd-power-identity}
    There is a set of coefficients $\coeffA{m}{0}, \coeffA{m}{1}, \ldots, \coeffA{m}{m}$ such that
    \begin{align*}
        n^{2m+1} = \sum_{r=0}^{m} \sum_{k=1}^{n} \coeffA{m}{r} k^r (n-k)^r
    \end{align*}
\end{theorem}
